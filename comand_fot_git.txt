cd directory

инициализация
git init

упаковка файлов
git add .

назначение комита
git commit -m "first commit" (пример)

подгруз с удалённого репозитория
git fetch name branch_name

аккуратное склеивание локального и удалённого репозиториев
git pull --rebase name branch_name

пуш
git push -u name branch_name(пример) -u только при первом

путь до репозитория
git remote add name https://gitlab.crja72.ru/golang/2025/spring/course/tasks/template.git (пример) только первый раз

ветка
git branch -M branch_name

просмотр удалённых репозиториев
git remote -v

удалени файла
 git rm --cached file_puth
